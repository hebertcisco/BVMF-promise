## Installation

```bash
npm i bvmf
```

### Usage

#### Import

```ts
//using ES6
import bvmf from 'bvmf';
//or using ES5
const bvmf  = require("bvmf")
```

#### Using

```ts
//using ES6
import bvmf from 'bvmf';

async function returnQuote(bvmf) {
    const result = await bvmf(
      {
        bvmf: bvmf
      });
    return(result);
 }
 try{
  console.log(returnQuote("itsa4"));
 }catch(err){
  console.error(err);
 }
```
